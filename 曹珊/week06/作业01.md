## 1、增

```python
from elasticsearch import Elasticsearch

# 1. 连接到本地 Elasticsearch（无认证）
es = Elasticsearch("http://127.0.0.1:9200")

# 2. 写入一条文档
doc = {
    "name": "Elasticsearch Python 操作",
    "price": 39.9,
    "tags": ["Python", "教程"]
}
resp = es.index(index="my_python_index", id=1, document=doc)
print("写入结果:", resp['result'])

# 3. 搜索文档
search_body = {
    "query": {
        "match": {
            "name": "Python"
        }
    }
}
resp = es.search(index="my_python_index", body=search_body)
print(resp)
print("找到", resp['hits']['total']['value'], "条结果")
for hit in resp['hits']['hits']:
    print(f"得分: {hit['_score']}, 数据: {hit['_source']}")
```
